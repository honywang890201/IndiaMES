                LEFT JOIN Bas_MO_Mac ON Bas_MO_Mac.LotId = Inp_Lot.LotId
                LEFT JOIN Bas_OP ON Bas_OP.OPId = Inp_Lot.OPId
                WHERE BAS_MO.MOCode = @MOCode and Bas_OP.OPCode = @OPCode
